#define Trig 7
#define Echo 8

void setup()
{
  Serial.begin(9600);
  
  pinMode(Trig, OUTPUT); //инициируем как выход 
  pinMode(Echo, INPUT); //инициируем как вход   
}

  /* Подаем импульс на вход trig дальномера */
